DROP TABLE Userinfo
DROP TABLE Blogpost
DROP TABLE Comment
CREATE TABLE Userinfo(authorid INT AUTO_INCREMENT PRIMARY KEY NOT NULL,username VARCHAR(30) NOT NULL,password VARCHAR(30) NOT NULL, firstname VARCHAR(30) NOT NULL,lastname VARCHAR(30) NOT NULL,email VARCHAR(50) NOT NULL,country VARCHAR(40),imageURL VARCHAR(100),status INTEGER DEFAULT 0)
CREATE TABLE Blogpost(postid INT AUTO_INCREMENT PRIMARY KEY NOT NULL,authorid INT NOT NULL,title VARCHAR(200),content TEXT, date DATE, category VARCHAR(20) DEFAULT "Off Topic")
CREATE TABLE Comment(commentid INT AUTO_INCREMENT PRIMARY KEY NOT NULL,postid INT NOT NULL, authorid INT NOT NULL,content TEXT, date DATE)
